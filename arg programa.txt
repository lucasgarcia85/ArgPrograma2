module Fideos
  @ajies = 0
    
  def self.ajies=(cantidad)
    @ajies=cantidad
  end
  
  def self.ajies
    @ajies
  end
  
  def self.recibir!
    @ajies+=5
  end
     
  def self.picantes?
    @ajies > 2
  end
  
  def self.descartar_la_salsa!
    if @ajies>10

  end
end
